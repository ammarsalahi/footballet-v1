def get_image(self,soup):
        img=soup.find(class_=self.site.img_tag)
        return img.contents[1]['src']

    def get_title(self,soup):
        return soup.title.string

    def get_body(self,soup):
        body=soup.find(class_=self.site.body_tag)
        bodytexts=[] 
        for b in body:
            bodytexts.append(b.text) 
        return bodytexts    

    def get_lead(self,soup):
        lead=soup.find(class_=self.site.lead_tag)
        leadtext=[] 
        for b in lead:
            leadtext.append(b.text) 
        return leadtext
